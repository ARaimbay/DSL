9.81e2=9.81 * 10 ^2
<expr> ::= <number> "*" <term> | <factor>
<number> ::= <integer> <fraction>
<fraction> ::= "." <digit>+
<integer> ::= <one-to-nine> {digit} | "0"
<digit> ::= "0" | <one_to_nine>
<one_to_nine> ::= "1"|"2"|"3"|"4"|"5"|"6"|"7"|"8"|"9"
<factor> ::= <digit> | <power> 
<power> ::= <base> "^" <exponent>
<base> ::= <one_to_nine> 
<exponent> ::= <digit>
